% Packages
\usepackage{longtable}                   % Breakable tables

% --------------------------
% GRAPHICS AND TABLE PACKAGES
% --------------------------
\usepackage{graphicx} % For including images (including logo)
\usepackage{array}    % Advanced columns in tables
\usepackage{booktabs} % Typographically better tables (\toprule, \midrule, \bottomrule)
\usepackage{enumitem} % For customizing list spacing

% --------------------------
% MATHEMATICAL PACKAGES
% --------------------------
\usepackage{amsmath,amssymb,amsthm} % Enhancements for mathematics (Theorems, symbols)
\usepackage{mathtools}              % Extensions to amsmath

% --------------------------
% NUMBERING OF FIGURES, TABLES, AND EQUATIONS
% --------------------------
% Makes numbering dependent on the chapter (Fig. 1.1, 1.2... ; Eq. (1.1), (1.2)...)
\numberwithin{figure}{chapter}
\numberwithin{table}{chapter}
\numberwithin{equation}{chapter}

% --------------------------
% HYPERREF FOR INTERNAL AND EXTERNAL LINKS
% --------------------------
% Create clickable links in PDF and set link colors
\usepackage{hyperref}
\hypersetup{
    colorlinks=true,
    linkcolor=black,
    citecolor=black,
    urlcolor=blue
}

% --------------------------
% SVG IMAGES
% --------------------------
\usepackage[inkscapeversion=1]{svg} % SVG images with Inkscape
